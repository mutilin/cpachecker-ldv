OBSERVER AUTOMATON PointerAnalysisObservingAutomaton
/* Queries the PointerAnalysis for errors and prints them.
 */

INITIAL STATE Init;

STATE USEFIRST Init :
  CHECK(PointerAnalysis,"DOUBLE_FREE") -> PRINT "Found a DOUBLE_FREE" ERROR;
  CHECK(PointerAnalysis,"MemoryLeak") -> PRINT "Observer: Found a Memory Leak" ERROR;
  CHECK(PointerAnalysis,"UNSAFE_DEREFERENCE") -> PRINT "Observer: Found an UNSAFE_DEREFERENCE" ERROR;
  CHECK(PointerAnalysis,"POTENTIALLY_UNSAFE_DEREFERENCE") -> PRINT "Observer: Found a POTENTIALLY_UNSAFE_DEREFERENCE" ERROR;
  CHECK(PointerAnalysis,"INVALID_FREE") -> PRINT "Observer: Found an INVALID_FREE" ERROR;

END AUTOMATON
